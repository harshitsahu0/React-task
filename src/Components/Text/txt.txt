      function handleclick() {
        // console.log(e);
        let value = Number(document.getElementById("inpt").value);
        // get all div
        if (value) {
          document.querySelectorAll(".chessboard div").forEach((x) => {
            let stringValue = value.toString();
            let valueNode = x.querySelector(".value");
            valueNode.innerHTML =
              stringValue.length < 5
                ? stringValue
                : stringValue.slice(0, 4) + "...";
            let tooltipNode = x.querySelector(".tooltip");
            tooltipNode.innerHTML = value;
            // if (stringValue.length < 5) {
            //   x.innerHTML = stringValue;
            // } else {
            //   x.innerHTML = stringValue.slice(0, 4) + "...";
            // }
            value = value * 2;
          });
        }
      }
